      </p>
      <pre style="background-color: #161b22; border: 1px solid #30363d; border-radius: 6px; padding: 12px; font-size: 11px; color: #c9d1d9; font-family: monospace; overflow-x: auto;">

window.addEventListener('resize', () => {
  document.body.classList.add('disable-blur'); // Supprime backdrop-filter

  resizeTimeout = setTimeout(() => {

});

useEffect(() => {
  const handler = (event, data) => setStats(data);
  window.opsecAPI.onSystemUpdate(handler);
  
  return () => {
    window.opsecAPI.removeSystemUpdateListener(handler); // Nettoyage strict
  };
}, []);

Napi::Value RunAuditAsync(const Napi::CallbackInfo& info) {
  Napi::Env env = info.Env();

  // Logique du worker thread natif...
